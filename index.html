const webhookUrl = 'https://discord.com/api/webhooks/YOUR_WEBHOOK_ID/YOUR_WEBHOOK_TOKEN'; function sendDiscordWebhook() { fetch(webhookUrl, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ content: 'buttonPressed' }) }) .then(response => { if (response.ok) alert('Triggered Discord webhook!'); else alert('Failed to trigger webhook.'); }) .catch(console.error); } document.getElementById('triggerBtn').onclick = sendDiscordWebhook;
